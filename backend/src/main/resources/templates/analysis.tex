\newcommand{\probbar}[2]{%
  \begin{tikzpicture}[baseline]
    \fill[lightgray] (0,0) rectangle (2.2,0.16);
    \pgfmathsetmacro{\w}{min(2.2,#1/100*2.2)}
    \fill[#2] (0,0) rectangle (\w,0.16);
  \end{tikzpicture}\hspace{1.0em}{\scriptsize #1\%}
}
